Users and Groups

group 'dba' do
  gid 500
end


user 'oracle' do
  uid 600
  gid 'dba'
end

group 'oinstall' do
  gid 501
  members 'oracle'
end

group 'oper' do
  gid 502
  members 'oracle'
end

Directories

  directory '/u01/app/oracle/product/10.2.0/db1' do
    owner 'oracle'
    group 'oinstall'
    mode '0755'
    recursive true
  end

Cron

cron "daily-report" do
  hour "5"
  minute "0"
  command "/usr/local/bin/daily-report"
end
